61 inline void tileRotate90(TileRect & tileRect, TileRect & tileRect90)
62 {
63  // x' = x*cos - y*sin
64  // y' = x*sin + y*cos
65  // where cos90 = 0 and sin90 = 1 (effectively clockwise)
66 
67  // rotate top right corner of rect
68  tileRect90.xmini = -tileRect.ymaxi;
69  tileRect90.ymini = tileRect.xmini;
70 
71  // swap width and height
72  tileRect90.xmaxi = tileRect90.xmini + (tileRect.ymaxi - tileRect.ymini);
73  tileRect90.ymaxi = tileRect90.ymini + (tileRect.xmaxi - tileRect.xmini);
74 }
