43  static constexpr std::tuple<double, double, double, bool> distanceFromLine(double cx, double cy, double ax, double ay, double bx, double by) noexcept {
44  // http://www.codeguru.com/forum/showthread.php?t=194400
45 
46  //
47  // find the distance from the point (cx,cy) to the line
48  // determined by the points (ax,ay) and (bx,by)
49  //
50 
51  double r_numerator = (cx-ax)*(bx-ax) + (cy-ay)*(by-ay);
52  double r_denomenator = (bx-ax)*(bx-ax) + (by-ay)*(by-ay);
53  double r = r_numerator / r_denomenator;
54  double dx = 0;
55  double dy = 0;
56  double distanceSegment = 0;
57  bool atEndpoint = false;
58 
59  if ( (r >= 0) && (r <= 1) )
60  {
61  dx = ax + r*(bx-ax);
62  dy = ay + r*(by-ay);
63  distanceSegment = (cx-dx)*(cx-dx) + (cy-dy)*(cy-dy);
64  atEndpoint = false;
65  }
66  else
67  {
68  atEndpoint = true;
69  double dist1 = (cx-ax)*(cx-ax) + (cy-ay)*(cy-ay);
70  double dist2 = (cx-bx)*(cx-bx) + (cy-by)*(cy-by);
71  if (dist1 < dist2)
72  {
73  dx = ax;
74  dy = ay;
75  distanceSegment = dist1;
76  }
77  else
78  {
79  dx = bx;
80  dy = by;
81  distanceSegment = dist2;
82  }
83  }
84 
85  return {dx, dy, distanceSegment, atEndpoint};
86  }
