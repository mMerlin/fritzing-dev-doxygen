43 /*
44  * A tile is the basic unit used for representing both space and
45  * solid area in a plane. It has the following structure:
46  *
47  * RT
48  * ^
49  * |
50  * -------------------------
51  * | | ---> TR
52  * | |
53  * | |
54  * | (lower left) |
55  * BL <--- -------------------------
56  * |
57  * v
58  * LB
59  *
60  * The (x, y) coordinates of the lower left corner of the tile are stored,
61  * along with four "corner stitches": RT, TR, BL, LB.
62  *
63  * Space tiles are distinguished at a higher level by having a distinguished
64  * tile body.
65  */
66

183 /*
184  * A plane of tiles consists of the four special tiles needed to
185  * surround all internal tiles on all sides. Logically, these
186  * tiles appear as below, except for the fact that all are located
187  * off at infinity.
188  *
189  * --------------------------------------
190  * |\ /|
191  * | \ / |
192  * | \ TOP / |
193  * | \ / |
194  * | \ / |
195  * | -------------------------- |
196  * | | | |
197  * |LEFT | |RIGHT|
198  * | | | |
199  * | -------------------------- |
200  * | / \ |
201  * | / \ |
202  * | / BOTTOM \ |
203  * | / \ |
204  * |/ \|
205  * --------------------------------------
206  */

220 /*
221  * The following coordinate, TINFINITY, is used to represent a
222  * tile location outside of the tile plane.
223  *
224  * It must be possible to represent TINFINITY+1 as well as
225  * INFINITY.
226  *
227  * Also, because locations involving TINFINITY may be transformed,
228  * it is desirable that additions and subtractions of small integers
229  * from either TINFINITY or MINFINITY not cause overflow.
230  *
231  * Consequently, we define TINFINITY to be the largest integer
232  * representable in wordsize - 2 bits.

306 /* The four macros below are for finding next tile RIGHT, UP, LEFT or DOWN
307  * from current tile at a given coordinate value.
308  *
309  * For example, NEXT_TILE_RIGHT points tResult to tile to right of t
310  * at y-coordinate y.
311  */
312 
313 
314 #define NEXT_TILE_RIGHT(tResult, t, y) \
315  for ((tResult) = TR(t); YMIN(tResult) > (y); (tResult) = LB(tResult)) \
316  /* Nothing */;
317 
318 #define NEXT_TILE_UP(tResult, t, x) \
319  for ((tResult) = RT(t); LEFT(tResult) > (x); (tResult) = BL(tResult)) \
320  /* Nothing */;
321 
322 #define NEXT_TILE_LEFT(tResult, t, y) \
323  for ((tResult) = BL(t); YMAX(tResult) <= (y); (tResult) = RT(tResult)) \
324  /* Nothing */;
325 
326 #define NEXT_TILE_DOWN(tResult, t, x) \
327  for ((tResult) = LB(t); RIGHT(tResult) <= (x); (tResult) = TR(tResult)) \
